üìã AS K2SF Framework Configuration Schema Preview

Each example includes a live editor for this configuration format:

{
  "as-md-datatable": {
    "extensionSettings": {
      "enabled": true
    },
    "templates": {
      "default": {
        "enabled": true,
        "autoGenerateColumns": false,
        "elevation": 2,
        "minHeight": 400,
        "optGrid": {
          "columns": [
            {
              "name": "columnName",
              "header": "Display Name", 
              "width": 150,
              "sortable": true,
              "editor": "text",
              "formatter": "currency"
            }
          ],
          "pageOptions": {
            "perPage": 10,
            "useClient": true
          },
          "selectionUnit": "row",
          "rowHeaders": ["rowNum", "checkbox"]
        }
      }
    },
    "targets": {
      "controls": [
        {
          "name": "GridControlName",
          "templates": "default"
        }
      ]
    }
  }
}

This matches the exact configuration structure used in your AS K2SF Framework solution.
